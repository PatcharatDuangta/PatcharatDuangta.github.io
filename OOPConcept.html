OOP Code Example in Java, C++, and Python

Java Code:

Java 

class Animal {
    Animal() {
        System.out.println("create animal");
    }

    protected void finalize() {
        System.out.println("delete animal");
    }

    void sound() {
        System.out.println("animal makes a sound");
    }
}

class Dog extends Animal {
    Dog() {
        System.out.println("dog");
    }

    @Override
    protected void finalize() {
        System.out.println("delete dog");
    }

    @Override
    void sound() {
        System.out.println("dog barks");
    }
}

class Cat extends Animal {
    Cat() {
        System.out.println("cat");
    }

    @Override
    protected void finalize() {
        System.out.println("delete cat");
    }

    @Override
    void sound() {
        System.out.println("cat meows");
    }
}

public class Main {
    public static void main(String[] args) {
        Animal a = new Animal();
        Dog d = new Dog();
        Cat c = new Cat();

        a.sound();
        d.sound();
        c.sound();

        a = null;
        d = null;
        c = null;
        System.gc();
    }
}

C++ Code:

C++ 

#include <iostream>
using namespace std;

class Animal {
public:
    Animal() {
        cout << "create animal" << endl;
    }
    
    virtual ~Animal() {
        cout << "delete animal" << endl;
    }

    virtual void sound() {
        cout << "animal makes a sound" << endl;
    }
};

class Dog : public Animal {
public:
    Dog() {
        cout << "dog" << endl;
    }

    ~Dog() {
        cout << "delete dog" << endl;
    }

    void sound() override {
        cout << "dog barks" << endl;
    }
};

class Cat : public Animal {
public:
    Cat() {
        cout << "cat" << endl;
    }

    ~Cat() {
        cout << "delete cat" << endl;
    }

    void sound() override {
        cout << "cat meows" << endl;
    }
};

int main() {
    Animal* a = new Animal();
    Animal* d = new Dog();
    Animal* c = new Cat();

    a->sound();
    d->sound();
    c->sound();

    delete a;
    delete d;
    delete c;

    return 0;
}

Python Code:

Python 

class Animal:
    def __init__(self):
        print("create animal")

    def __del__(self):
        print("delete animal")

    def sound(self):
        print("animal makes a sound")


class Dog(Animal):
    def __init__(self):
        super().__init__()
        print("dog")

    def __del__(self):
        print("delete dog")

    def sound(self):
        print("dog barks")


class Cat(Animal):
    def __init__(self):
        super().__init__()
        print("cat")

    def __del__(self):
        print("delete cat")

    def sound(self):
        print("cat meows")


if __name__ == "__main__":
    a = Animal()
    d = Dog()
    c = Cat()

    a.sound()
    d.sound()
    c.sound()

    del a
    del d
    del c

ข้อ II: การทำความเข้าใจส่วนประกอบของโค้ด

1. Class (คลาส)

คลาสคือแบบพิมพ์ (blueprint) สำหรับการสร้างออบเจ็กต์ ซึ่งเป็นการรวบรวมข้อมูล (attributes) และพฤติกรรม (methods) ไว้ในรูปแบบที่สามารถนำไปใช้สร้างออบเจ็กต์ใหม่ได้ ตัวอย่างในโค้ดคือ:

class Animal

คลาส Animal ในโค้ดเป็นแบบพิมพ์สำหรับสัตว์ทั่วไป ที่มีเมธอด sound เพื่อแสดงพฤติกรรมในการส่งเสียง

2. Object (ออบเจ็กต์)

ออบเจ็กต์คืออินสแตนซ์ของคลาส เมื่อเราสร้างออบเจ็กต์ เรากำลังสร้างตัวตนที่มีข้อมูลและพฤติกรรมตามที่กำหนดในคลาส ตัวอย่างเช่น:

Animal a = new Animal();

ออบเจ็กต์ a เป็นอินสแตนซ์ของคลาส Animal ซึ่งมีคุณสมบัติและพฤติกรรมที่กำหนดไว้ในคลาสนั้น

3. Subclass (ซับคลาส)

ซับคลาสคือคลาสที่สืบทอดคุณสมบัติและพฤติกรรมจากคลาสแม่ (superclass) โดยสามารถปรับเปลี่ยนพฤติกรรมหรือเพิ่มคุณสมบัติใหม่ได้ ตัวอย่างเช่น คลาส Dog และ Cat ที่สืบทอดจากคลาส Animal:

class Dog extends Animal
class Cat extends Animal

ทั้ง Dog และ Cat เป็นซับคลาสของ Animal โดยที่ทั้งสองสามารถปรับเปลี่ยนพฤติกรรมของเมธอด sound ได้ตามลักษณะเฉพาะของตนเอง

4. Message (การส่งข้อความ)

การส่งข้อความใน OOP คือการเรียกใช้งานเมธอดของออบเจ็กต์ เมื่อเราสร้างออบเจ็กต์แล้ว เราสามารถส่งข้อความเพื่อสั่งให้มันทำบางสิ่งบางอย่าง ตัวอย่างเช่น:

d.sound();

คำสั่งนี้เป็นการส่งข้อความไปยังออบเจ็กต์ d (Dog) เพื่อให้เรียกใช้เมธอด sound ของมัน ซึ่งในกรณีนี้จะพิมพ์ "dog barks"

5. Inheritance (การสืบทอด)

Inheritance หรือการสืบทอดคือการที่ซับคลาสรับคุณสมบัติและพฤติกรรมจากคลาสแม่โดยไม่ต้องเขียนโค้ดซ้ำ ตัวอย่างเช่น Dog และ Cat สืบทอดพฤติกรรมพื้นฐานของ Animal แต่สามารถปรับเปลี่ยนเมธอด sound ได้

6. Polymorphism (พอลิมอร์ฟิซึม)

Polymorphism คือความสามารถของออบเจ็กต์ในการใช้เมธอดเดียวกันแต่มีพฤติกรรมที่แตกต่างกันในซับคลาส ตัวอย่างเช่น แม้ว่า Dog และ Cat จะมีเมธอด sound เหมือนกัน แต่พฤติกรรมของเมธอดในแต่ละคลาสจะแตกต่างกัน:

d.sound(); // Output: dog barks
c.sound(); // Output: cat meows

นี่คือการแสดงให้เห็นถึงการใช้ polymorphism ในการกำหนดพฤติกรรมที่แตกต่างกันของออบเจ็กต์ที่มาจากซับคลาสต่าง ๆ
