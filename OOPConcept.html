เปรียบเทียบการเขียนโปรแกรมแบบ OOP ใน Java, C++, Python

ความเหมือน

  • ทั้งสามภาษาเป็นภาษาที่สนับสนุนการเขียนโปรแกรมเชิงวัตถุ (Object-Oriented Programming - OOP).
  • สนับสนุนหลักการของ OOP ได้แก่:
    • การสืบทอด (Inheritance)
    • การห่อหุ้ม (Encapsulation)
    • การหลายรูปแบบ (Polymorphism)
    • การแบ่งแยกหน้าที่ (Abstraction)

ความแตกต่าง

Java

- มีการจัดการหน่วยความจำแบบอัตโนมัติ (Garbage Collection).

- สนับสนุนการทำงานแบบ cross-platform ผ่าน JVM.

C++

- ให้ผู้เขียนโค้ดจัดการหน่วยความจำเอง (manual memory management) ผ่าน pointer.

- รองรับทั้งการเขียนแบบ procedural และ OOP.

Python

- เป็นภาษาที่มี syntax ที่เรียบง่ายและเข้าใจง่าย.

- ไม่มีการประกาศประเภทตัวแปร (dynamic typing).

ตัวอย่างโค้ดในแต่ละภาษา

Java


class VendingMachine {
    private String[] items = {"Snack", "Drink", "Candy"};
    private int[] prices = {50, 20, 10};
    private int balance = 0;

    public void insertMoney(int amount) {
        balance += amount;
        System.out.println("Inserted: " + amount + " Current balance: " + balance);
    }

    public void selectItem(int itemNumber) {
        if (itemNumber < 0 || itemNumber >= items.length) {
            System.out.println("Invalid selection.");
            return;
        }
        if (balance >= prices[itemNumber]) {
            balance -= prices[itemNumber];
            System.out.println("Dispensed: " + items[itemNumber] + " Remaining balance: " + balance);
        } else {
            System.out.println("Not enough balance. Please insert more money.");
        }
    }

    public void returnChange() {
        System.out.println("Returning change: " + balance);
        balance = 0;
    }
}

public class Main {
    public static void main(String[] args) {
        VendingMachine vm = new VendingMachine();
        vm.insertMoney(50);
        vm.selectItem(0);
        vm.returnChange();
    }
}

C++


#include <iostream>
#include <string>
using namespace std;

class VendingMachine {
private:
    string items[3] = {"Snack", "Drink", "Candy"};
    int prices[3] = {50, 20, 10};
    int balance = 0;

public:
    void insertMoney(int amount) {
        balance += amount;
        cout << "Inserted: " << amount << " Current balance: " << balance << endl;
    }

    void selectItem(int itemNumber) {
        if (itemNumber < 0 || itemNumber >= 3) {
            cout << "Invalid selection." << endl;
            return;
        }
        if (balance >= prices[itemNumber]) {
            balance -= prices[itemNumber];
            cout << "Dispensed: " << items[itemNumber] << " Remaining balance: " << balance << endl;
        } else {
            cout << "Not enough balance. Please insert more money." << endl;
        }
    }

    void returnChange() {
        cout << "Returning change: " << balance << endl;
        balance = 0;
    }
};

int main() {
    VendingMachine vm;
    vm.insertMoney(50);
    vm.selectItem(0);
    vm.returnChange();
    return 0;
}

Python


class VendingMachine:
    def __init__(self):
        self.items = ["Snack", "Drink", "Candy"]
        self.prices = [50, 20, 10]
        self.balance = 0

    def insert_money(self, amount):
        self.balance += amount
        print(f"Inserted: {amount} Current balance: {self.balance}")

    def select_item(self, item_number):
        if item_number < 0 or item_number >= len(self.items):
            print("Invalid selection.")
            return
        if self.balance >= self.prices[item_number]:
            self.balance -= self.prices[item_number]
            print(f"Dispensed: {self.items[item_number]} Remaining balance: {self.balance}")
        else:
            print("Not enough balance. Please insert more money.")

    def return_change(self):
        print(f"Returning change: {self.balance}")
        self.balance = 0

# Example usage
vm = VendingMachine()
vm.insert_money(50)
vm.select_item(0)
vm.return_change()

สรุปการเปรียบเทียบ

ลักษณะ Java C++ Python
Syntax ค่อนข้างเป็นทางการ มีโครงสร้างที่ชัดเจน ยืดหยุ่น สามารถเขียนได้ทั้งแบบ OOP และ procedural เรียบง่าย และเข้าใจได้ง่าย
การจัดการหน่วยความจำ จัดการอัตโนมัติด้วย Garbage Collection จัดการด้วยตัวเอง (manual) ผ่าน pointer จัดการอัตโนมัติด้วย Garbage Collection
การเข้าถึงข้อมูล ใช้ private, protected, public ใช้ private, protected, public ทุกอย่างเป็น public โดยปกติ แต่สามารถใช้ _ หรือ __ เพื่อบอกว่าเป็น private
การใช้งาน Class จำเป็นต้องสร้าง Class ในทุกๆ การทำงานแบบ OOP ยืดหยุ่นกว่าในการเขียนได้ทั้งแบบ procedural และ OOP ใช้ Class ได้อย่างยืดหยุ่นและเข้าใจง่าย
